c. pentru un graf reprezentat cu matricea de adiacenta sa se determine matricea distantelor.

#include <iostream>
#include<fstream>
#define infinit 100000
using namespace std;
ifstream f("date.in");
int a[105][105], m[105][105];
int main()
{int n, ok=0, j, i, k;
f>>n;
for(i=1;i<=n;i++)
    for(j=1;j<=n;j++)
        f>>a[i][j];
//Se foloseste algoritmul lui Warshall
//initializam matricea distantelor
for(i=1;i<=n;i++)
    for(j=1;j<=n;j++)
         { if (i == j)
                m[i][j] = 0;
            if (a[i][j] == 0 && i != j)
                m[i][j] = infinit;
            if (a[i][j] == 1)
                m[i][j] = a[i][j];
         }
//algoritmul lui warshall
for (k = 1; k <= n; k++)
        for (i = 1; i <= n; i++)
            for (j = 1; j <= n; j++)
                if (m[i][k]!=infinit && m[k][j]!=infinit)
                if (m[i][j] > m[i][k] + m[k][j])
                    m[i][j] = m[i][k] + m[k][j];

for(i=1;i<=n;i++)
    {for(j=1;j<=n;j++)
        cout<<m[i][j]<<" ";
    cout<<endl;
}
    return 0;
}
